function getTodayDiscretionarySpend() {
  const today = new Date().toDateString();
  const NON_DISC = ['Debt repayment','Income','Savings','Bills','Transfer','Loan'];
  return S.txns
    .filter(t => {
      const d = new Date(t.ts || t.date);
      return d.toDateString() === today && !t.income && !NON_DISC.includes(t.cat);
    })
    .reduce((s,t) => s + (t.amt||0), 0);
}

        const nameEl = row.querySelector('.row-name, .bill-name');
        if (!nameEl) continue;
        const name = nameEl.textContent.trim();
        const bill = BILLS.find(b => b.name === name);
        if (bill && isThisMonthlyBillPaid(bill.name, bill.day)) return false;
      }
      return true;
    } catch(e) { return true; }
  }
},
{
  name: 'Over by alert excludes debt repayments',
  check: () => {
    try {
      const overEl = document.querySelector('[id*="overspend"]');
      if (!overEl || !overEl.textContent.includes('Over by')) return true;
      const displayed = parseFloat(overEl.textContent.match(/\$([\d,.]+)/)?.[1]?.replace(',','') || '0');
      const discretionary = getTodayDiscretionarySpend ? getTodayDiscretionarySpend() : 0;
      const expected = Math.max(0, discretionary - getDailyBudget());
      return Math.abs(displayed - expected) < 2;
    } catch(e) { return true; }
  }
},
